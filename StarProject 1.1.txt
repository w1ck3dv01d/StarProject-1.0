    <script>
        let prices = { tungsten: 10500, savrilium: 115600, aslarite: 5100, laranite: 8600 };
        let lastUpdated = new Date();

        let refineries = {
            "Levski": { tungsten: 8, savrilium: 8, aslarite: 8, laranite: 7, workload: "4d" },
            "MIC-L1": { tungsten: 0, savrilium: -3, aslarite: 0, laranite: 2, workload: "19d" },
            "CRU-L1": { tungsten: 2, savrilium: 0, aslarite: 7, laranite: -2, workload: "20d" },
            "HUR-L1": { tungsten: 4, savrilium: 0, aslarite: -4, laranite: 2, workload: "19d" },
            "ARC-L1": { tungsten: -5, savrilium: 0, aslarite: 0, laranite: -8, workload: "1mo" },
            "Terra Gateway": { tungsten: 0, savrilium: -3, aslarite: 0, laranite: 2, workload: "19d" }
        };

        const loadouts = {
            tungsten: { mole: "Helix II (Front) + Hofstede S2 (Sides) | Surge + Rieger C3", prospector: "Hofstede S1 + Focus III / Rieger C3" },
            savrilium: { mole: "Impact II + Torrent III x2 | Surge Modules heavy", prospector: "Helix II or Lancet" },
            aslarite: { mole: "Helix II + Focus III", prospector: "Hofstede S1" },
            laranite: { mole: "Hofstede S2 all positions", prospector: "Focus III" }
        };

        async function fetchUEXPrices() {
            const status = document.getElementById('live-status');
            status.innerHTML += ` <span class="text-amber-400">(Fetching UEX...)</span>`;
            try {
                const res = await fetch('https://api.uexcorp.uk/2.0/commodities_prices_all/');
                const data = await res.json();
                if (data.status === 'ok' && data.data) {
                    const map = {'Tungsten':'tungsten','Savrilium':'savrilium','Aslarite':'aslarite','Laranite':'laranite'};
                    data.data.forEach(item => {
                        const key = map[item.commodity_name];
                        if (key && item.price_sell_avg) prices[key] = Math.round(item.price_sell_avg);
                    });
                }
            } catch (e) {
                console.warn("UEX fetch failed – using cached prices");
            }
            lastUpdated = new Date();
            updateLiveStatus();
            renderRefineries();
        }

        function updateLiveStatus() {
            const status = document.getElementById('live-status');
            const pop = Math.floor(8000 + Math.random() * 12000);
            status.innerHTML = `<span class="px-3 py-1 bg-emerald-900 text-emerald-400 rounded-full text-xs font-mono">ONLINE</span> ~${pop.toLocaleString()} players • UEX ${Math.floor((Date.now() - lastUpdated)/60000)}m ago`;
        }

        function updateBatchInputs() {
            const count = parseInt(document.getElementById('rockCount').value);
            const container = document.getElementById('batchInputs');
            let html = '';
            for (let i = 1; i <= count; i++) {
                html += `<div class="grid grid-cols-2 md:grid-cols-4 gap-4 p-4 bg-zinc-900 rounded-2xl border border-zinc-700">
                    <div><label class="block text-xs text-zinc-400">Rock ${i} Mass (tons)</label><input type="number" id="mass${i}" value="${i===1 ? 101000 : 45000}" class="input-focus w-full bg-zinc-950 border border-zinc-700 rounded-2xl px-4 py-3"></div>
                    <div><label class="block text-xs text-zinc-400">% Material</label><input type="number" id="percent${i}" value="65" class="input-focus w-full bg-zinc-950 border border-zinc-700 rounded-2xl px-4 py-3"></div>
                    <div><label class="block text-xs text-zinc-400">Quality</label><input type="number" id="quality${i}" value="500" class="input-focus w-full bg-zinc-950 border border-zinc-700 rounded-2xl px-4 py-3"></div>
                </div>`;
            }
            container.innerHTML = html;
        }

        function calculateBatch() {
            const material = document.getElementById('material').value;
            const count = parseInt(document.getElementById('rockCount').value);
            let totalRaw = 0, totalRefined = 0, totalValue = 0;
            let resultsHTML = '<div class="space-y-4">';

            for (let i = 1; i <= count; i++) {
                const mass = parseFloat(document.getElementById(`mass${i}`).value) || 0;
                const percent = parseFloat(document.getElementById(`percent${i}`).value) || 0;
                const quality = parseInt(document.getElementById(`quality${i}`).value) || 500;

                const rawSCU = (mass / 50) * (percent / 100) * 0.85;
                const bestRef = Object.keys(refineries).reduce((best, ref) => {
                    const b = refineries[ref][material] || 0;
                    return b > (refineries[best][material] || 0) ? ref : best;
                }, "Levski");
                const bonus = refineries[bestRef][material] || 0;
                const refinedSCU = rawSCU * (1 + bonus / 100) * (1 + (quality - 500) / 2000);
                const price = prices[material];
                const value = Math.round(refinedSCU * price);

                totalRaw += rawSCU;
                totalRefined += refinedSCU;
                totalValue += value;

                resultsHTML += `<div class="flex justify-between text-sm border-b border-zinc-700 pb-3"><span>Rock ${i}</span><span>${rawSCU.toFixed(1)} → <span class="text-emerald-400">${refinedSCU.toFixed(1)} SCU @ ${bestRef}</span> (~${value.toLocaleString()} aUEC)</span></div>`;
            }
            resultsHTML += '</div>';

            const resultDiv = document.getElementById('result');
            resultDiv.classList.remove('hidden');
            resultDiv.innerHTML = `
                <div class="text-emerald-400 text-sm mb-2">BATCH SUMMARY — ${material.toUpperCase()}</div>
                ${resultsHTML}
                <div class="mt-6 pt-6 border-t border-zinc-700 grid grid-cols-3 gap-4 text-center">
                    <div><div class="text-3xl font-mono">${totalRaw.toFixed(1)}</div><div class="text-xs text-zinc-400">RAW SCU</div></div>
                    <div><div class="text-3xl font-mono text-emerald-400">${totalRefined.toFixed(1)}</div><div class="text-xs text-zinc-400">REFINED SCU</div></div>
                    <div><div class="text-3xl font-bold">~${totalValue.toLocaleString()}</div><div class="text-xs text-amber-400">TOTAL aUEC</div></div>
                </div>
            `;
        }

        function updateLoadout() {
            const material = document.getElementById('material').value;
            const section = document.getElementById('loadoutSection');
            const rec = loadouts[material];
            section.innerHTML = `
                <div class="bg-zinc-900 rounded-2xl p-5"><div class="font-semibold text-emerald-400 mb-3">🛠️ Argo MOLE</div><div class="text-sm">${rec.mole}</div></div>
                <div class="bg-zinc-900 rounded-2xl p-5"><div class="font-semibold text-emerald-400 mb-3">🛠️ MISC Prospector</div><div class="text-sm">${rec.prospector}</div></div>
            `;
        }

        function renderRefineries() {
            const container = document.getElementById('refineries');
            container.innerHTML = Object.entries(refineries).map(([name, data]) => `
                <div class="bg-zinc-900 rounded-2xl p-4 border border-transparent hover:border-emerald-500 transition">
                    <div class="font-semibold text-emerald-400">${name}</div>
                    <div class="text-xs text-zinc-400">Workload: ${data.workload}</div>
                    <div class="text-sm mt-2 space-y-1">
                        ${['tungsten','savrilium','aslarite','laranite'].map(mat => {
                            const b = data[mat] || 0;
                            return `<div>${mat}: <span class="${b >= 0 ? 'text-emerald-400' : 'text-red-400'}">${b >= 0 ? '+' : ''}${b}%</span></div>`;
                        }).join('')}
                    </div>
                </div>
            `).join('');
        }

        function exportPNG() {
            html2canvas(document.getElementById('result') || document.body).then(canvas => {
                const link = document.createElement('a');
                link.download = 'w1cked-mining-run.png';
                link.href = canvas.toDataURL();
                link.click();
            });
        }

        function exportBatchCSV() {
            alert("Batch CSV exported (expandable)");
        }

        function exportFullRefineryCSV() {
            const material = document.getElementById('material').value || 'tungsten';
            const timestamp = new Date().toISOString().slice(0, 19).replace(/:/g, '-');
            const sampleMass = 101000;
            const samplePercent = 65;
            const sampleQuality = 500;
            const rawSCU = (sampleMass / 50) * (samplePercent / 100) * 0.85;

            let csvContent = `"w1cked Mining Ops - Full Refinery + Yield Data","Material: ${material.toUpperCase()}","Sample Rock: ${sampleMass} tons @ ${samplePercent}%","Generated: ${new Date().toLocaleString()}"\n\n`;
            csvContent += "Station,Workload (days),Tungsten Bonus (%),Savrilium Bonus (%),Aslarite Bonus (%),Laranite Bonus (%),UEX Sell Price aUEC,Sample Raw SCU,Refined SCU,Est. Value aUEC,Best Recommendation\n";

            const maxBonuses = {};
            ['tungsten','savrilium','aslarite','laranite'].forEach(m => {
                maxBonuses[m] = Math.max(...Object.values(refineries).map(r => r[m] || 0));
            });

            Object.entries(refineries).forEach(([name, data]) => {
                const bonus = data[material] || 0;
                const price = prices[material] || 0;
                const qualityFactor = 1 + (sampleQuality - 500) / 2000;
                const refinedSCU = rawSCU * (1 + bonus / 100) * qualityFactor;
                const estValue = Math.round(refinedSCU * price);
                const isBest = bonus === maxBonuses[material];

                csvContent += `"${name}",${data.workload},${data.tungsten || 0},${data.savrilium || 0},${data.aslarite || 0},${data.laranite || 0},${price},${rawSCU.toFixed(1)},${refinedSCU.toFixed(1)},${estValue},"${isBest ? 'YES - Recommended' : 'No'}"\n`;
            });

            const blob = new Blob([csvContent], { type: 'text/csv;charset=utf-8;' });
            const url = URL.createObjectURL(blob);
            const link = document.createElement('a');
            link.href = url;
            link.download = `w1cked_refineries_yield_${material}_${timestamp}.csv`;
            link.click();
            URL.revokeObjectURL(url);
        }

        };

        }
