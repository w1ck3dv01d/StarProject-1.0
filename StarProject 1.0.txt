    <script>
        // Refinery data
        const refineries = {
            "MIC-L1": { tungsten: 0, savrilium: -3, aslarite: 0 },
            "CRU-L1": { tungsten: 2, quantanium: 7 },
            "HUR-L1": { tungsten: 4, laranite: 8 },
            "Levski": { tungsten: 8, savrilium: 8, aslarite: 8, laranite: 6 },
            "ARC-L1": { aluminum: 7, tungsten: -5 }
        };

        // Sample price data (updated via "scrape" simulation)
        let prices = {
            tungsten: 11200,
            savrilium: 9800,
            aslarite: 8500,
            laranite: 6700
        };

        function updateLiveStatus() {
            const status = document.getElementById('live-status');
            const hour = new Date().getHours();
            const pop = Math.floor(8000 + Math.random() * 9000);
            status.innerHTML = `
                <span class="px-3 py-1 bg-emerald-900 text-emerald-400 rounded-full text-xs font-mono">ONLINE</span>
                ~${pop.toLocaleString()} players
            `;
        }

        async function refreshStats() {
            const container = document.getElementById('server-stats');
            container.innerHTML = `
                <div class="animate-pulse space-y-3">
                    <div class="h-6 bg-zinc-800 rounded"></div>
                    <div class="h-6 bg-zinc-800 rounded w-3/4"></div>
                </div>
            `;
            
            // Simulate API call
            setTimeout(() => {
                container.innerHTML = `
                    <div class="space-y-3 text-sm">
                        <div class="flex justify-between"><span class="text-zinc-400">Concurrent Players</span><span class="font-mono text-emerald-400">~12,450</span></div>
                        <div class="flex justify-between"><span class="text-zinc-400">Active Shards</span><span class="font-mono">47</span></div>
                        <div class="flex justify-between"><span class="text-zinc-400">Avg per Shard</span><span class="font-mono">265</span></div>
                        <div class="flex justify-between"><span class="text-zinc-400">Best for Mining</span><span class="text-amber-400">EU Low-Pop</span></div>
                    </div>
                `;
            }, 800);
        }

        function calculateYield() {
            const mass = parseFloat(document.getElementById('mass').value) || 0;
            const percent = parseFloat(document.getElementById('percent').value) || 0;
            const material = document.getElementById('material').value;
            const quality = parseInt(document.getElementById('quality').value) || 500;

            const rawSCU = (mass / 50) * (percent / 100) * 0.85;
            const bestRef = Object.keys(refineries).reduce((best, ref) => {
                const bonus = refineries[ref][material] || 0;
                return bonus > (refineries[best][material] || 0) ? ref : best;
            }, "Levski");
            const bonus = refineries[bestRef][material] || 0;
            const refinedSCU = rawSCU * (1 + bonus / 100);
            const price = prices[material] || 10000;
            const value = Math.round(refinedSCU * price);

            const resultDiv = document.getElementById('result');
            resultDiv.classList.remove('hidden');
            resultDiv.innerHTML = `
                <div class="grid grid-cols-2 gap-6">
                    <div>
                        <div class="text-emerald-400 text-sm">RAW YIELD</div>
                        <div class="text-4xl font-mono">${rawSCU.toFixed(1)} <span class="text-xl">SCU</span></div>
                    </div>
                    <div>
                        <div class="text-emerald-400 text-sm">REFINED @ ${bestRef}</div>
                        <div class="text-4xl font-mono">${refinedSCU.toFixed(1)} <span class="text-xl">SCU (+${bonus}%)</span></div>
                    </div>
                </div>
                <div class="mt-6 pt-6 border-t border-zinc-700">
                    <div class="text-amber-400 text-sm">EST. VALUE</div>
                    <div class="text-5xl font-bold text-white">~${value.toLocaleString()} <span class="text-2xl">aUEC</span></div>
                    <div class="text-xs text-zinc-500 mt-1">Material: ${material.toUpperCase()} @ ${price} aUEC/SCU • Quality: ${quality}</div>
                </div>
                <div class="mt-4 text-xs text-zinc-400">💡 Target low-pop shards + ${bestRef} for best runs.</div>
            `;
        }

        function renderRefineries() {
            const container = document.getElementById('refineries');
            container.innerHTML = Object.entries(refineries).map(([name, bonuses]) => `
                <div class="bg-zinc-900 rounded-2xl p-4">
                    <div class="font-semibold text-emerald-400">${name}</div>
                    <div class="text-xs text-zinc-400 mt-2">Bonuses:</div>
                    <div class="text-sm mt-1 space-y-0.5">
                        ${Object.entries(bonuses).map(([mat, b]) => 
                            `<div>${mat}: <span class="${b >= 0 ? 'text-emerald-400' : 'text-red-400'}">+${b}%</span></div>`
                        ).join('')}
                    </div>
                </div>
            `).join('');
        }

        // Init
        window.onload = () => {
            updateLiveStatus();
            setInterval(updateLiveStatus, 45000);
            refreshStats();
            renderRefineries();
            
            // Auto-calculate on input change
            document.querySelectorAll('input, select').forEach(el => {
                el.addEventListener('change', calculateYield);
            });
            calculateYield(); // initial calc
        };
    </script>
